Week 1 Plus Minus
#include <assert.h>
#include <ctype.h>
#include <limits.h>
#include <math.h>
#include <stdbool.h>
#include <stddef.h>
#include <stdint.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void plusMinus(int arr[], int n){
    int pos = 0, neg = 0, zero = 0;
    
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0)
            pos++;
        else if (arr[i] < 0)
            neg++;
        else
            zero++;
    }

    printf("%.6f\n", (float)pos / n);
    printf("%.6f\n", (float)neg / n);
    printf("%.6f\n", (float)zero / n);
}

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    plusMinus(arr, n);

    return 0;
}


Week 1 Mini-Max Sum

#include <assert.h>
#include <ctype.h>
#include <limits.h>
#include <math.h>
#include <stdbool.h>
#include <stddef.h>
#include <stdint.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#include <stdio.h>

void miniMaxSum(int arr[5]) {
    long long total = 0;
    int min = arr[0];
    int max = arr[0];

    for (int i = 0; i < 5; i++) {
        total += arr[i];

        if (arr[i] < min) {
            min = arr[i];
        }

        if (arr[i] > max) {
            max = arr[i];
        }
    }

    long long minSum = total - max;
    long long maxSum = total - min;

    printf("%lld %lld\n", minSum, maxSum);
}

int main() {
    int arr[5];

    // Read 5 space-separated integers
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }

    miniMaxSum(arr);

    return 0;
}

Week 1 Time Conversion

#include <stdio.h>
#include <string.h>
#include <stdlib.h>

void timeConversion(char* s) {
    int hour;
    char time[9]; // HH:MM:SS
    char period[3]; // AM or PM

    // Extract hour, minutes, seconds, and AM/PM
    sscanf(s, "%2d:%*2d:%*2d%2s", &hour, period);  // Read hour and period (AM/PM)

    // Adjust hour based on AM/PM
    if (strncmp(s + 8, "AM", 2) == 0) {
        if (hour == 12) hour = 0;
    } else {
        if (hour != 12) hour += 12;
    }

    // Print formatted 24-hour time
    printf("%02d:%.2s:%.2s\n", hour, s + 3, s + 6);
}

int main() {
    char s[11];  // e.g., "07:05:45PM"
    scanf("%s", s);
    timeConversion(s);
    return 0;
}
